# suppress std error
Remove-Item not-found.abc 2>&1 | Out-Null
Remove-Item not-found.abc -ErrorAction Ignore

# only supress std error
Remove-Item not-found.abc 2>$null

# redirect stdout/stderr to variable $var1, suppress output to terminal
Write-Error 'abc' 2>&1 | Tee-Object -Variable var1 | Out-Null
$var1

# redirect stdout/stderr to variable $var2, suppress output to terminal
$var2 = Write-Error 'abc' 2>&1
$var2

$var3 = Write-Error 'abc' *> $null
$var3
