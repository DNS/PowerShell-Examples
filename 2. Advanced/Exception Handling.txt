
try {
	Get-Content test.txt
} catch {
	# catch only catches terminating errors, this will not get executed
	# some ‘errors’ may be non-terminating errors and these do not get caught.
	Write-Output 'Catch Error'
} finally {
	Write-Output 'Finally Error'
}




try {
	Get-Content test.txt -ErrorAction Stop
} catch {
	# `-ErrorAction Stop` will caught error
	Write-Output 'Catch Error'
} finally {
	Write-Output 'Finally Error'
}
